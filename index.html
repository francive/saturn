// --- CORRECCIÓN: ANILLO CON AGUJERO (Mantiene el brillo pero no atraviesa el planeta) --- // Usamos RingGeometry(radioInterno, radioExterno, segmentos) // El radio interno de 1.4 es suficiente para que el planeta (1.2) pase por el medio. const glowRingGeo = new THREE.RingGeometry(1.4, 5, 64); // El material se queda EXACTAMENTE igual para no perder tu efecto de color const glowRingMat = new THREE.ShaderMaterial({ uniforms: { color: { value: new THREE.Color(0xFFD700) } }, vertexShader: ` varying vec2 vUv; void main() { vUv = uv; gl_Position = projectionMatrix * modelViewMatrix * vec4( position, 1.0 ); } `, fragmentShader: ` uniform vec3 color; varying vec2 vUv; void main() { // Nota: En RingGeometry, las UVs se distribuyen de forma distinta, // pero el efecto de distancia seguirá funcionando para el brillo. float dist = distance(vUv, vec2(0.5, 0.5)); float alpha = smoothstep(0.5, 0.2, dist) * 0.6; vec3 finalColor = mix(vec3(1.0, 0.8, 0.3), color, smoothstep(0.3, 0.5, dist)); gl_FragColor = vec4(finalColor, alpha); } `, transparent: true, depthWrite: false, blending: THREE.AdditiveBlending }); const glowRing = new THREE.Mesh(glowRingGeo, glowRingMat); // Ya no necesitamos rotarlo en X -90 porque RingGeometry se crea "de frente" // y el ringGroup ya tiene la inclinación de 15 grados que te gusta. // Si ves que queda "parado", deja esta línea: glowRing.rotation.x = 0; // O bórrala, ya que el grupo manda. glowRing.position.y = -0.05; ringGroup.add(glowRing);
